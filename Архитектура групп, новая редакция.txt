3.5.1 Бэкенд архитектура (backend) групп
backend/lib/modules/groups/
├── models/
│   ├── training_group.model.dart     # Модель тренировочной группы (TrainingGroup)
│   ├── analytic_group.model.dart     # Модель аналитической группы (AnalyticGroup)
│   ├── group_schedule_slot.model.dart # Модель слота расписания группы
│   ├── group_condition.model.dart    # Модель условия для аналитических групп
│   └── group_member.model.dart       # Модель участника группы (связь пользователя с группой)
├── controllers/
│   ├── training_groups.controller.dart # Контроллер для тренировочных групп (CRUD API)
│   ├── analytic_groups.controller.dart # Контроллер для аналитических групп (CRUD API)
│   ├── group_schedule.controller.dart  # Контроллер для управления расписанием групп
│   └── group_members.controller.dart   # Контроллер для управления участниками групп
├── services/
│   ├── training_group.service.dart     # Бизнес-логика тренировочных групп
│   ├── analytic_group.service.dart     # Бизнес-логика аналитических групп
│   ├── group_schedule.service.dart     # Сервис работы с расписанием групп
│   ├── group_member.service.dart       # Сервис управления участниками
│   └── group_sync.service.dart         # Сервис синхронизации аналитических групп
├── repositories/
│   ├── training_group.repository.dart  # Репозиторий для работы с БД (training_groups)
│   ├── analytic_group.repository.dart  # Репозиторий для аналитических групп
│   ├── group_schedule.repository.dart  # Репозиторий для слотов расписания
│   └── group_member.repository.dart    # Репозиторий для участников групп
└── validators/
    ├── training_group.validator.dart   # Валидаторы данных тренировочных групп
    ├── analytic_group.validator.dart   # Валидаторы аналитических групп
    └── group_schedule.validator.dart   # Валидаторы расписания групп
